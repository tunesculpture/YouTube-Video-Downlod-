# main.py — Your FastAPI backend

from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import FileResponse
from pydantic import BaseModel
import yt_dlp, os, uuid

app = FastAPI()

app.add_middleware(CORSMiddleware,
  allow_origins=["*"],
  allow_methods=["*"],
  allow_headers=["*"])

class DownloadRequest(BaseModel):
  url: str
  format: str = "mp4"
  quality: str = "720p"

@app.get("/info")
async def get_info(url: str):
  with yt_dlp.YoutubeDL({}) as ydl:
    info = ydl.extract_info(url, download=False)
    return {"title": info["title"],
     "thumbnail": info["thumbnail"],
     "duration": info["duration"],
     "channel": info["uploader"]}

@app.post("/download")
async def download_video(req: DownloadRequest):
  filename = f"/tmp/{uuid.uuid4()}.%(ext)s"
  opts = {"outtmpl": filename,
    "format": "bestaudio" if req.format in ["mp3","m4a","wav","ogg"] else "bestvideo+bestaudio"}
  if req.format == "mp3":
    opts["postprocessors"] = [{"key": "FFmpegExtractAudio", "preferredcodec": "mp3"}]
  with yt_dlp.YoutubeDL(opts) as ydl:
    ydl.download([req.url])
  return FileResponse(path, filename="download")
# Start the backend server
uvicorn main:app --host 0.0.0.0 --port 8000 --reload
